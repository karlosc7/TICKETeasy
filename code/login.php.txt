    $sql = "SELECT * FROM usuarios WHERE usuario = '$user' and password = '$password'";

    $count = 0;

    while($row = mysql_fetch_object($rec))
    {
        $count++;
        $result = $row;
    }

    if($count == 1)
    {
        return 1;
    }

    {
        return 0;
    }
